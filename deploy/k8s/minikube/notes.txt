PREREQS: A working installation of `kubectl` and `minikube`

# Do it all
  (minikube delete || true) && \
  minikube start --memory 8192 && \
  eval $(minikube docker-env) && \
  mvn clean package docker:build && \
  kubectl create -f deploy/k8s/minikube && \
  kubectl get pods && \
  echo '*** NOTE ***: When `kubectl get pods` shows everything is running, run `minikube service nginx-ingress` to open the LB in your browser'


# Or piece-by-piece

# Reset minikube

  $ minikube delete || true

# Start minikube with 8GB memory

  $ minikube start --memory 8192

# Setup environment

    $ eval $(minikube docker-env)

# Build docker images on minikube

  $ mvn clean package docker:build

# Load everything

    $ kubectl create -f deploy/k8s/minikube

# Open ingress in browser

    $ minikube service nginx-ingress
