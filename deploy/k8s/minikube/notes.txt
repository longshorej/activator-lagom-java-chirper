NOTE: I'm using this file to archive some commands and general workflow. Its contents
      will ultimately end up somewhere else.

# Build images

    $ mvn clean package docker:build

# Reset minikube

  $ minikube delete
  $ minikube start

# Setup environment

    $ eval $(minikube docker-env)

# Load everything

    $ kubectl create -f deploy/k8s/minikube

# Open ingress in browser

    $ minikube service nginx-ingress
