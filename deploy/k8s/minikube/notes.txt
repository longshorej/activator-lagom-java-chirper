# Create a stateful set for front-end

    $ kubectl create -f deploy/k8/minikube/lagom/front-end/front-end-statefulset.json
    statefulset "front-end" created
    -> 0

    $ kubectl get statefulsets
    NAME        DESIRED   CURRENT   AGE
    front-end   1         0         15s
    -> 0

# Expose pod-0 and open it in the browser

    $ kubectl expose pod front-end-0 --type=LoadBalancer
    service "front-end-0" exposed
    -> 0

     $ kubectl get services
    NAME          CLUSTER-IP   EXTERNAL-IP   PORT(S)           AGE
    front-end-0   10.0.0.174   <pending>     12002:31089/TCP   11s
    kubernetes    10.0.0.1     <none>        443/TCP           19h
    -> 0

    $ minikube service front-end-0
    Opening kubernetes service default/front-end-0 in default browser...
    -> 0
